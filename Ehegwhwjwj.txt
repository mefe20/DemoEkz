d) Убедиться, что сервер отдаёт course_name и phone

В вашем серверном файле (bookingController.js) запрос getAllBookings должен выглядеть так:

```js
exports.getAllBookings = async (req, res) => {
    try {
        const bookings = await pool.query(`
            SELECT b.id, u.fio, u.phone, c.name as course_name, b.start_date, b.status, b.payment_method, b.review
            FROM Bookings b 
            JOIN Users u ON b.user_id = u.id 
            JOIN Courses c ON b.cours_id = c.id
        `);
        res.status(200).json(bookings.rows);
    } catch (err) {
        res.status(500).json({ error: err.message });
    }
};
```
